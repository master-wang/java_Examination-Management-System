用户表

CREATE TABLE IF NOT EXISTS `user`(
`id` INT UNSIGNED AUTO_INCREMENT,
`username` VARCHAR(60) NULL,
`password` VARCHAR(60)  NULL,
`name` VARCHAR(60)  NULL,
`phone` VARCHAR(60)  NULL,
`headImg` VARCHAR(300)  NULL,
`idCardNo` VARCHAR(60)  NULL,
`is_admin` VARCHAR(60)  NULL DEFAULT 'false',
PRIMARY KEY ( `id` )
)ENGINE=InnoDB DEFAULT CHARSET=utf8;

考博列表
CREATE TABLE IF NOT EXISTS `test_list`(
`T_id` INT UNSIGNED AUTO_INCREMENT,
`school` VARCHAR(100) NULL,
`is_open` VARCHAR(60)  NULL DEFAULT 'false',
 project varchar(100) NULL,
PRIMARY KEY ( `T_id` )
)ENGINE=InnoDB DEFAULT CHARSET=utf8;

考博列表科目
CREATE TABLE IF NOT EXISTS `test_list_item`(
`I_id` INT UNSIGNED AUTO_INCREMENT,
`T_id` INT,
`title` varchar(100) NULL,
PRIMARY KEY ( `I_id` )
)ENGINE=InnoDB DEFAULT CHARSET=utf8;

考博列表科目題目
CREATE TABLE IF NOT EXISTS `test_list_item_title`(
`i_id` INT UNSIGNED AUTO_INCREMENT,
`Item_id` INT,
`title` VARCHAR(200)  NULL,
`answer` VARCHAR(60)  NULL,
`secA` VARCHAR(10)  NULL,
`secB` VARCHAR(10)  NULL,
`secC` VARCHAR(10)  NULL,
PRIMARY KEY ( `i_id` )
)ENGINE=InnoDB DEFAULT CHARSET=utf8;

成绩
CREATE TABLE IF NOT EXISTS `grade`(
`g_id` INT UNSIGNED AUTO_INCREMENT,
`u_id` INT,
`i_id` INT,
`grade` VARCHAR(30)  NULL,
`mianshi` VARCHAR(30)  NULL DEFAULT 'false',
PRIMARY KEY ( `g_id` )
)ENGINE=InnoDB DEFAULT CHARSET=utf8;
